///////////////////GraphQL Update Mutation With Connect//////////////////
mutation {
  updateBusinesses(
    where: { businessId_EQ: "b10" }
    update: {
      categories: {
        connect: [ {
          where: { node: { OR: [{ name_EQ: "Restaurant" }, { name_EQ: "Breakfast" }] } }
        }]
      }
    }
  ) {
    businesses {
      name
      categories {
        name
      }
    }
  }
}
